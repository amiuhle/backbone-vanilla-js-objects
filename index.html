Backbone: Vanilla JS Objects

Display plain JavaScript Objects with Backbone.js

Installation

$ bower install backbone-vanilla-js-objects --save-dev

Examples

All examples use the following variable assignments:

var ObjectView = Backbone.VanillaJsObjects.Views.Object;
var ArrayView = Backbone.VanillaJsObjects.Views.Array;
var PropertyView = Backbone.VanillaJsObjects.Views.Property;

Object

Create a Backbone.VanillaJsObjects.Views.Object and give it any JavaScript object that should be displayed:

$('#example-1').append(
  new ObjectView({
    inspect: {
      string: 'bar',
      number: 1,
      boolean: true,
      null: null,
      undefined: undefined,
      function: function() {},
      array: [
        'text',
        123,
        {
          I: 'Am an Object'
        }
      ],
      object: {
        aLot: Infinity,
        nan: NaN
      }
    }
  }).render().el
);

Arrays

$('#example-2').append(
  new ArrayView({
    inspect: [
      'text',
      1.1,
      false,
      {
        foo: 'bar'
      },
      window.close
    ]
  }).render().el
);

Top-level properties

$('#example-3').append(
  new PropertyView({
    tagName: 'div',
    model: new Backbone.VanillaJsObjects.Property({
      property: 'underscore',
      value: _
    })
  }).render().el
);
